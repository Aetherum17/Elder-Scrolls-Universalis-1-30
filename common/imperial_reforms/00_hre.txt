#8 ADM - Procalim Absolute Integrity 
es_proclaim_absolute_integrity = {
	empire = hre
	potential = {
		NOT = { has_dlc = "Emperor" }
	}
	province = {
		supply_limit_modifier = 0.15
	}
	member = {
		spy_offence = 0.25
	}
	emperor = {
		free_leader_pool = 1
	}
	on_effect = {
		custom_tooltip = "ES_EMPEROR_ANNEX_PRINCES"
		hidden_effect = {
		change_tag = TAM 
		if = {
			limit = {
				NOT = { government_rank = 10 }
			}
			set_government_rank = 10
		}
		set_country_flag = formed_tam_flag
		add_country_modifier = {
			name = "centralization_modifier"
			duration = 7300
		}
		add_prestige = 25
		if = { limit = { has_custom_ideas = no } country_event = { id = ideagroups.1 } }
			every_country = {
				limit = {
					is_part_of_hre = yes
					is_emperor = no
					num_of_cities = 1
					has_opinion = { who = ROOT value = 0 }
					reverse_has_opinion = { who = ROOT value = 0 }
					ai = yes
				}
				emperor = { inherit = ROOT }
			}
		}
	}
	required_reform = es_emperor_surpress_independent_opposition
	gui_container = nodlc
}
