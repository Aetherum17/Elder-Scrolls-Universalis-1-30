	}

		name = institution_events.2.a

		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 20

			limit = {
				NOT = { has_completed_all_reforms_trigger = yes }
			}

		FROM = {
			add_permanent_province_modifier = {
				name = "birthplace_of_the_renaissance"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = institution_events.200 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					NOT = { knows_country = ROOT }
				}
				country_event = { id = institution_events.201 }
			}
		}

country_event = {
	id = institution_events.8
	title = institution_events.8.t
	desc = institution_events.8.d
	picture = HORDE_ON_HORSEBACK_eventPicture

	is_triggered_only = yes
	
	trigger = {
		NOT = { has_institution = feudalism }
		any_owned_province = {
			feudalism = 100
		}
		any_owned_province = {
			NOT = { feudalism = 50 }
			feudalism = 25
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = institution_events.8.a
		random_owned_province = {
			limit = {
				NOT = { feudalism = 50 }
				feudalism = 25
			}
			add_institution_embracement = {
				value = 50
				which = "feudalism"
			}
			spawn_rebels = {
				size = 1
				type = noble_rebels
			}
		}
		if = {
			limit = {
				has_estate = estate_nomadic_tribes
			}
			add_estate_loyalty = {
				estate = estate_nomadic_tribes
				loyalty = -10
			}
		}
	}
	option = {
		name = institution_events.8.b
		random_owned_province = {
			limit = {
				NOT = { feudalism = 100 }
				feudalism = 25
			}
			add_institution_embracement = {
				value = -25
				which = "feudalism"
			}
		}
	}
}

#Growing Urban Influence
country_event = {
	id = institution_events.9
	title = institution_events.9.t
	desc = institution_events.9.d
	picture = CITY_VIEW_eventPicture

	is_triggered_only = yes
	
	trigger = {
		NOT = { has_institution = feudalism }
		any_owned_province = {
			feudalism = 100
		}
		any_owned_province = {
			is_capital = no
			NOT = { feudalism = 50 }
			development = 10
			base_manpower = 2
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					development = 10
					base_manpower = 2
					is_capital = no
					NOT = { feudalism = 50 }
				}
				set_province_flag = growing_urbanism
			}
		}
	}

	option = {
		name = institution_events.9.b
		random_owned_province = {
			limit = {
				has_province_flag = growing_urbanism
			}
			clr_province_flag = growing_urbanism
			add_institution_embracement = {
				value = 50
				which = "feudalism"
			}
			add_base_manpower = -1
		}
	}
	option = {
		name = institution_events.9.a
		add_horde_unity = 5
		add_legitimacy = 5
		add_republican_tradition = 2.5
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = growing_urbanism
				}
				clr_province_flag = growing_urbanism
			}
		}
	}
}
