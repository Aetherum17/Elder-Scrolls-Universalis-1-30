namespace = et

#Mission to Moon: Launch
country_event = {
	id = et.90
	title = et.EVTNAME90
	desc = et.EVTDESC90
	picture = ES_SPACE_1_eventPicture
	
	is_triggered_only = yes

	option = {
		name = et.EVTOPTA90
		add_prestige = 5
		if = {
			limit = { NOT = { has_global_flag = man_on_moon } }
			country_event = {
				id = et.91
				days = 8
			}
		}
		if = {
			limit = { has_global_flag = man_on_moon }
			country_event = {
				id = et.92
				days = 8
			}
		}
		set_global_flag = man_on_moon
	}
}
